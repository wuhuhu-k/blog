# 实现

const createTree = function (list = []) {
  const tree = [];
  const map = {};
  const waitPushObj = {};
  list.forEach((item) => {
    const newItem = {
      id: item.id,
      pid: item.pid,
      name: item.name,
      code: item.code,
      children: [],
    };

    map[item.id] = newItem;
    // 判断是否是根节点
    if (item.pid === -1) {
      tree.push(newItem);
    } else if (map[item.pid]) {
      map[item.pid].children.push(newItem);
    } else {
      if (!waitPushObj[item.pid]) {
        waitPushObj[item.pid] = [];
      }
      waitPushObj[item.pid].push(newItem);
    }

    // 判断暂存区是否有自己的子节点
    if (waitPushObj[item.id]) {
      newItem.children = waitPushObj[item.id];
      delete waitPushObj[item.id];
    }

    // 可以不加这个判断
    if (!map[item.id]) {
      map[item.id] = newItem;
    }
  });
  return tree;
};
